import React, { useEffect, useRef, useState } from 'react';
import { LinearProgress } from '@mui/material';

const YouTubeVideoWithProgress = () => {
  const [player, setPlayer] = useState(null);
  const [progress, setProgress] = useState(0);
  const playerRef = useRef(null);

  // Load YouTube IFrame API script
  useEffect(() => {
    const loadScript = () => {
      const script = document.createElement('script');
      script.src = "https://www.youtube.com/iframe_api";
      document.body.appendChild(script);
    };

    if (!window.YT) {
      loadScript();
    }

    window.onYouTubeIframeAPIReady = () => {
      setPlayer(new window.YT.Player(playerRef.current, {
        height: '360',
        width: '640',
        videoId: 'YOUR_VIDEO_ID', // Replace with your video ID
        events: {
          'onStateChange': onPlayerStateChange,
        },
      }));
    };
  }, []);

  // Update progress based on video state
  useEffect(() => {
    const interval = setInterval(() => {
      if (player) {
        const duration = player.getDuration();
        const currentTime = player.getCurrentTime();
        const progress = (currentTime / duration) * 100;
        setProgress(progress);
      }
    }, 1000);

    return () => clearInterval(interval);
  }, [player]);

  // Handle player state changes (e.g., play, pause)
  const onPlayerStateChange = (event) => {
    if (event.data === window.YT.PlayerState.ENDED) {
      setProgress(0);
    }
  };

  return (
    <div>
      <div ref={playerRef}></div>
      <LinearProgress variant="determinate" value={progress} />
    </div>
  );
};

export default YouTubeVideoWithProgress;
